# рекурсивная функция быстрой сортировки
def quik_sort(s):
    # если длина переданного списка <= 1 возвращаем список
    if len(s) <= 1:
        return s
    # опорный элемент (любой, например первый)
    elem = s[0]
    # список чисел меньших опорного элемента
    left = [i for i in s if i < elem]
    # список чисел равных опорному элементу
    middle = [i for i in s if i == elem]
    # список чисел больших опорного элемента
    rigth = [i for i in s if i > elem]
    # возврящает склеенный список из вызова функции для левой части + центр + вызова функции для правой части
    return quik_sort(left) + middle + quik_sort(rigth)
